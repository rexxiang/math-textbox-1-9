\section*{基本运算符号}


\begin{center}
\begin{tabular}{lll}
\toprule
符号 & 含义 & 示例 \\
\midrule
$+$ & 加 & $3 + 5 = 8$ \\
$-$ & 减 & $8 - 3 = 5$ \\
$\times$ & 乘 & $4 \times 6 = 24$ \\
$\div$ & 除 & $24 \div 6 = 4$ \\
$=$ & 等于 & $2 + 3 = 5$ \\
$\neq$ & 不等于 & $3 \neq 5$ \\
$\approx$ & 约等于 & $\pi \approx 3.14$ \\
\bottomrule
\end{tabular}
\end{center}
