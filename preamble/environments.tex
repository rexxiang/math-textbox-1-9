%% ── Auxiliary commands ────────────────────────────────────────────
% 前置知识（inline, yellow tint）
\newcommand{\prerequisite}[1]{%
  \begin{tcolorbox}[
    size=small, colback=yellow!10, colframe=yellow!55,
    leftrule=2pt, sharp corners, boxsep=2pt,
  ]
  \small\textbf{前置知识：} #1
  \end{tcolorbox}}

% 适用年级（inline, teal tint）
\newcommand{\gradelevel}[1]{%
  \begin{tcolorbox}[
    size=small, colback=teal!7, colframe=teal!40,
    leftrule=2pt, sharp corners, boxsep=2pt,
  ]
  \small\textbf{适用年级：} #1
  \end{tcolorbox}}

%% ── Theorem-style environments (amsthm) ──────────────────────────

\newtheorem{definition}[theorem]{定义}
\newtheorem{property}[theorem]{性质}
\newtheorem{corollary}[theorem]{推论}
\newtheorem{lemma}[theorem]{引理}
